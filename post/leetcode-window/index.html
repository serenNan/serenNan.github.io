bool checkInclusion(string t, string s)
{

class Solution {
public:
    int lengthOfLongestSubstring(string s) 
    {
        unordered_map<char, int> window,need;
        int len = 0;

        int left = 0, right = 0;
        while(right < s.size())
        {
            char c = s[right];
            window[c]++;
            right++;
            while(window[c] > 1)
            {
                char d = s[left];
                left++;
                window[d]--;
                
            }
            len = max(len,right - left);
        }
        return len;
    }
};
